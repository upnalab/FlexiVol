68  _tClassType = asm.GetType(PluginTypeName);

72  Debug.LogError("Voxon Runtime failed to load from local C#-Runtime.dll.");
73  Windows.Error("Voxon Runtime failed to load from local C#-Runtime.dll.");
74  _runtime = null;

92  catch (Exception e)

101  private void FindDll()
102  {
103  if (_pluginFilePath != "") return;
104 
105  if (File.Exists(PluginFileName))
106  {
107  _pluginFilePath = PluginFileName;
108  return;
109  }
110 
111 #if NET_4_6
112  RegistryKey dll = Registry.LocalMachine.OpenSubKey("SOFTWARE\\Voxon\\Voxon");
113  if (dll != null)
114  {
115  _pluginFilePath =
116  $"{(string) Registry.LocalMachine.OpenSubKey("SOFTWARE\\Voxon\\Voxon")?.GetValue("Path")}{PluginFileName}";
117  return;
118  }
119 #endif
120 
121  string voxon_path = Environment.GetEnvironmentVariable("VOXON_RUNTIME", EnvironmentVariableTarget.User);
122 
123  if (File.Exists($"{voxon_path}{PluginFileName}"))
124  {
125  _pluginFilePath = $"{voxon_path}{PluginFileName}";
126  return;
127  }
128  else
129  {
130  Debug.Log($"{voxon_path}{PluginFileName} Doesn't Exist");
131 
132  }
133 
134  string[] paths = Environment.GetEnvironmentVariable("Path", EnvironmentVariableTarget.User)?.Split(';');
135 
136  if (paths == null) return;
137 
138  foreach (string path in paths)
139  {
140  if (!File.Exists($"{path}\\{PluginFileName}")) continue;
141 
142  _pluginFilePath = $"{path}\\{PluginFileName}";
143  }
