179  private void rearrange_indices(ref Mesh mesh)
180  {
181  /* Triangles are 3 idx and our array requires -1 delimiting,
182  / So we need to make room for all tris (count) + a -1 at the end of each (count / 3)
183  */
184  try
185  {
186  for (var submesh = 0; submesh < submeshCount; submesh++)
187  {
188  int indicesCount = mesh.GetTriangles(submesh).Length;
189  indexCounts[submesh] = indicesCount + (indicesCount / 3);
190 
191  indices[submesh] = new int[indexCounts[submesh]]; // Number of Poly Indices
192 
193  // Set up indices
194  int[] triMe = mesh.GetTriangles(submesh);
195 
196  var outIdx = 0;
197  for (var i = 0; i < indicesCount; i += 3, outIdx += 4)
198  {
199  // Copy internal array to output array
200  indices[submesh][0 + outIdx] = triMe[i + 0];
201  indices[submesh][1 + outIdx] = triMe[i + 1];
202  indices[submesh][2 + outIdx] = triMe[i + 2];
203 
204  // flag end of triangle
205  indices[submesh][3 + outIdx] = -1;
206  }
207  }
208  }
209  catch (Exception e)
210  {
