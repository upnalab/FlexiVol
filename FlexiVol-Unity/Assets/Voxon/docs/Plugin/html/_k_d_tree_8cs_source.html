170  KDBounds MakeBounds() {
171 
172  Vector3 max = new Vector3(float.MinValue, float.MinValue, float.MinValue);
173  Vector3 min = new Vector3(float.MaxValue, float.MaxValue, float.MaxValue);
174 
175  int even = Count & ~1; // calculate even Length
176 
177  // min, max calculations
178  // 3n/2 calculations instead of 2n
179  for (int i0 = 0; i0 < even; i0 += 2) {
180 
181  int i1 = i0 + 1;
182 
183  // X Coords
184  if (points[i0].x > points[i1].x) {
185  // i0 is bigger, i1 is smaller
186  if (points[i1].x < min.x)
187  min.x = points[i1].x;
188 
189  if (points[i0].x > max.x)
190  max.x = points[i0].x;
191  }
192  else {
193  // i1 is smaller, i0 is bigger
194  if (points[i0].x < min.x)
195  min.x = points[i0].x;
196 
197  if (points[i1].x > max.x)
198  max.x = points[i1].x;
199  }
200 
201  // Y Coords
202  if (points[i0].y > points[i1].y) {
203  // i0 is bigger, i1 is smaller
204  if (points[i1].y < min.y)
205  min.y = points[i1].y;
206 
207  if (points[i0].y > max.y)
208  max.y = points[i0].y;
209  }
210  else {
211  // i1 is smaller, i0 is bigger
212  if (points[i0].y < min.y)
213  min.y = points[i0].y;
214 
215  if (points[i1].y > max.y)
216  max.y = points[i1].y;
217  }
218 
219  // Z Coords
220  if (points[i0].z > points[i1].z) {
221  // i0 is bigger, i1 is smaller
222  if (points[i1].z < min.z)
223  min.z = points[i1].z;
224 
225  if (points[i0].z > max.z)
226  max.z = points[i0].z;
227  }
228  else {
229  // i1 is smaller, i0 is bigger
230  if (points[i0].z < min.z)
231  min.z = points[i0].z;
232 
233  if (points[i1].z > max.z)
234  max.z = points[i1].z;
235  }
236  }
237 
238  // if array was odd, calculate also min/max for the last element
239  if(even != Count) {
240  // X
241  if (min.x > points[even].x)
242  min.x = points[even].x;
243 
244  if (max.x < points[even].x)
245  max.x = points[even].x;
246  // Y
247  if (min.y > points[even].y)
248  min.y = points[even].y;
249 
250  if (max.y < points[even].y)
251  max.y = points[even].y;
252  // Z
253  if (min.z > points[even].z)
254  min.z = points[even].z;
255 
256  if (max.z < points[even].z)
257  max.z = points[even].z;
258  }
259 
260  KDBounds b = new KDBounds();
261  b.min = min;
262  b.max = max;
263 
264  return b;
265  }

273  void SplitNode(KDNode parent) {
274 
275  // center of bounding box
276  KDBounds parentBounds = parent.bounds;
277  Vector3 parentBoundsSize = parentBounds.size;
278 
279  // Find axis where bounds are largest
280  int splitAxis = 0;
281  float axisSize = parentBoundsSize.x;
282 
283  if (axisSize < parentBoundsSize.y) {
284  splitAxis = 1;
285  axisSize = parentBoundsSize.y;
286  }
287 
288  if (axisSize < parentBoundsSize.z) {
289  splitAxis = 2;
290  }
291 
292  // Our axis min-max bounds
293  float boundsStart = parentBounds.min[splitAxis];
294  float boundsEnd = parentBounds.max[splitAxis];
295 
296  // Calculate the spliting coords
297  float splitPivot = CalculatePivot(parent.start, parent.end, boundsStart, boundsEnd, splitAxis);
298 
299  parent.partitionAxis = splitAxis;
300  parent.partitionCoordinate = splitPivot;
301 
302  // 'Spliting' array to two subarrays
303  int splittingIndex = Partition(parent.start, parent.end, splitPivot, splitAxis);
304 
305  // Negative / Left node
306  Vector3 negMax = parentBounds.max;
307  negMax[splitAxis] = splitPivot;
308 
309  KDNode negNode = GetKDNode();
310  negNode.bounds = parentBounds;
311  negNode.bounds.max = negMax;
312  negNode.start = parent.start;
313  negNode.end = splittingIndex;
314  parent.negativeChild = negNode;
315 
316  // Positive / Right node
317  Vector3 posMin = parentBounds.min;
318  posMin[splitAxis] = splitPivot;
319 
320  KDNode posNode = GetKDNode();
321  posNode.bounds = parentBounds;
322  posNode.bounds.min = posMin;
323  posNode.start = splittingIndex;
324  posNode.end = parent.end;
325  parent.positiveChild = posNode;
326 
327  // check if we are actually splitting it anything
328  // this if check enables duplicate coordinates, but makes construction a bit slower
329 #if KDTREE_DUPLICATES
330  if(negNode.Count != 0 && posNode.Count != 0) {
331  #endif
332  // Constraint function deciding if split should be continued
333  if(ContinueSplit(negNode))
334  SplitNode(negNode);
335 
336 
337  if(ContinueSplit(posNode))
338  SplitNode(posNode);
339 
340 #if KDTREE_DUPLICATES
341  }
342 #endif
343  }

359  float CalculatePivot(int start, int end, float boundsStart, float boundsEnd, int axis) {
360 
362  float midPoint = (boundsStart + boundsEnd) / 2f;
363 
364  bool negative = false;
365  bool positive = false;
366 
367  float negMax = Single.MinValue;
368  float posMin = Single.MaxValue;
369 
370  // this for loop section is used both for sorted and unsorted data
371  for (int i = start; i < end; i++) {
372 
373  if (points[permutation[i]][axis] < midPoint)
374  negative = true;
375  else
376  positive = true;
377 
378  if (negative == true && positive == true)
379  return midPoint;
380  }
381 
382  if (negative) {
383 
384  for (int i = start; i < end; i++)
385  if (negMax < points[permutation[i]][axis])
386  negMax = points[permutation[i]][axis];
387 
388  return negMax;
389  }
390  else {
391 
392  for (int i = start; i < end; i++)
393  if (posMin > points[permutation[i]][axis])
394  posMin = points[permutation[i]][axis];
395 
396  return posMin;
397  }
398  }

415  int Partition(int start, int end, float partitionPivot, int axis) {
416 
417  // note: increasing right pointer is actually decreasing!
418  int LP = start - 1; // left pointer (negative side)
419  int RP = end; // right pointer (positive side)
420 
421  int temp; // temporary var for swapping permutation indexes
422 
423  while (true) {
424 
425  do {
426  // move from left to the right until "out of bounds" value is found
427  LP++;
428  }
429  while (LP < RP && points[permutation[LP]][axis] < partitionPivot);
430 
431  do {
432  // move from right to the left until "out of bounds" value found
433  RP--;
434  }
435  while (LP < RP && points[permutation[RP]][axis] >= partitionPivot);
436 
437  if (LP < RP) {
438  // swap
439  temp = permutation[LP];
440  permutation[LP] = permutation[RP];
441  permutation[RP] = temp;
442  }
443  else {
444 
445  return LP;
446  }
447  }
448  }
